if (!admin.apps.length) {

**FILE 3: service/backend/src/controllers/signals.controller.js**

Structure and logic:

```js
'use strict';
const { getFirestore } = require('../config/firebase.admin');
const { VirtualPortfolio } = require('../services/virtualPortfolio.service');
const marketDataService = require('../services/marketData.service');

// In-memory portfolio registry: strategyId -> VirtualPortfolio instance
// Each strategy config (from Firestore) gets its own portfolio.
// This resets on server restart — acceptable for forward-test mode.
const _portfolios = new Map();

function _getOrCreatePortfolio(strategyId, initialCapital = 1000000) {
  if (!_portfolios.has(strategyId)) {
    _portfolios.set(strategyId, new VirtualPortfolio(initialCapital));
  }
  return _portfolios.get(strategyId);
}

// Fetch strategy config from Firestore signal_strategies/{token}
async function _lookupStrategy(token) {
  const db = getFirestore();
  if (!db) return null;
  const doc = await db.collection('signal_strategies').doc(token).get();
  if (!doc.exists) return null;
  const data = doc.data();
  if (!data.isActive) return null;
  return { id: doc.id, ...data };
}

// Persist signal log to Firestore signal_log
async function _logSignal(logEntry) {
  const db = getFirestore();
  if (!db) return null;
  try {
    const ref = await db.collection('signal_log').add({
      ...logEntry,
      receivedAt: require('firebase-admin').firestore.FieldValue.serverTimestamp(),
    });
    return ref.id;
  } catch (err) {
    console.error('[SignalsController] Failed to log signal:', err.message);
    return null;
  }
}
```

**`receiveTradingView(req, res)` handler:**

Validation:
- Required fields: `token`, `action`, `symbol`. If any missing, return `res.status(400).json({ error: 'Missing required fields', required: ['token', 'action', 'symbol'], received: Object.keys(req.body) })`
- Normalize: `action = req.body.action.toUpperCase()` — must be one of `['BUY', 'SELL', 'EXIT']`, else return `400 { error: 'Invalid action. Must be BUY, SELL, or EXIT' }`
- `token = req.body.token` (string)
- `symbol = req.body.symbol` (string)
- `quantity = parseInt(req.body.quantity, 10) || 50` (default 50 if not provided)

Strategy lookup:
```js
const strategy = await _lookupStrategy(token);
if (!strategy) {
  await _logSignal({ source: 'tradingview', token, symbol, action, quantity, status: 'rejected', fillPrice: null, errorMessage: 'Token not found or strategy inactive', mode: 'unknown' });
  return res.status(401).json({ error: 'Invalid webhook token or strategy not active' });
}
```

Execution in paper mode (`strategy.mode === 'paper'`):
```js
const portfolio = _getOrCreatePortfolio(strategy.id);
const ltp = marketDataService.lastPrice[symbol] || 0;
if (ltp <= 0) {
  return res.status(503).json({ error: 'No live price available for ' + symbol + '. Is feed connected?' });
}

let fillResult;
try {
  if (action === 'BUY') {
    fillResult = portfolio.buy(symbol, quantity, ltp, strategy.slippage || 0.001);
  } else if (action === 'SELL' || action === 'EXIT') {
    fillResult = portfolio.sell(symbol, quantity, ltp, strategy.slippage || 0.001);
  }
} catch (err) {
  await _logSignal({ source: 'tradingview', token, strategyId: strategy.id, symbol, action, quantity, status: 'error', fillPrice: null, errorMessage: err.message, mode: 'paper' });
  return res.status(422).json({ error: err.message });
}
```

Execution in live mode (`strategy.mode === 'live'`):
- Return `501 { error: 'Live trading not yet implemented', mode: 'live' }` — stub for Phase 2.

Log and respond:
```js
await _logSignal({
  source: 'tradingview', token, strategyId: strategy.id, symbol, action,
  quantity, status: 'filled', fillPrice: fillResult?.fillPrice || ltp,
  errorMessage: null, mode: strategy.mode,
});
return res.status(200).json({
  status: 'ok',
  source: 'tradingview',
  strategy: strategy.strategyName,
  symbol,
  action,
  fill: fillResult,
  portfolio: portfolio.getSummary(marketDataService.lastPrice),
});
```

**`receiveChartink(req, res)` handler:**

Parse Chartink payload:
```js
const { stocks, trigger_prices, triggered_at, scan_name } = req.body;
if (!stocks || !trigger_prices) {

const symbolList = stocks.split(',').map(s => s.trim());
const priceList = trigger_prices.split(',').map(p => parseFloat(p.trim()));

if (symbolList.length !== priceList.length) {
  return res.status(400).json({ error: 'stocks and trigger_prices arrays must be same length' });

```

Process each symbol:
```js
const results = [];
for (let i = 0; i < symbolList.length; i++) {
  const symbol = symbolList[i];
  const triggerPrice = priceList[i];
  results.push({
    symbol,
    triggerPrice,
    scan_name: scan_name || 'unknown',
    triggered_at: triggered_at || new Date().toISOString(),
    status: 'received',
  });
  await _logSignal({
    source: 'chartink',
    token: 'chartink_' + (scan_name || 'unknown'),
    strategyId: null,
    symbol,
    action: 'ALERT',
    quantity: 0,
    status: 'received',
    fillPrice: triggerPrice,
    errorMessage: null,
    mode: 'paper',
    scan_name,
    triggered_at,
  });
}
return res.status(200).json({ status: 'ok', source: 'chartink', processed: results.length, signals: results });
```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Internet → POST /api/signals/tradingview | Public endpoint; any actor can POST if URL is known |
| Internet → POST /api/signals/chartink | No auth on Chartink endpoint by design — Chartink sends without headers |
| Webhook token → Firestore strategy | Token is the authentication credential; must not be guessable |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Spoofing | POST /api/signals/tradingview | mitigate | Token lookup in Firestore — invalid token returns 401; token must be UUID (unguessable) |
| T-03-02 | Repudiation | Signal execution | mitigate | Every signal (including rejected ones) written to signal_log with timestamp and token |
| T-03-03 | Information Disclosure | signal_log Firestore | accept | Contains fill prices and strategy names — internal collection, no public API exposes it |
| T-03-04 | Denial of Service | Chartink flood | accept | Chartink endpoint rate is bounded by Chartink's own trigger rate; no auth needed as it has no execution path (mode: paper/alert only) |
| T-03-05 | Elevation of Privilege | live mode bypass | mitigate | Live mode returns 501 stub — no broker order can be placed in Phase 1; enforced in controller |
| T-03-06 | Tampering | quantity field | mitigate | `parseInt(qty, 10) || 50` — NaN/negative qty falls to default; VirtualPortfolio throws on qty <= 0 |
</threat_model>
